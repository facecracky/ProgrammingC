#include <stdio.h>

int main() {
    int N = 3;
    double arr[N][N];
    for(int i = 0; i < N; i++) {
        for(int j = 0; j < N; j++) {
            scanf("%lf", &arr[i][j]);
        }
    }

    for(int i = 0; i < N; i++) {
        for(int j = 0; j < N; j++) {
            printf("%lf ", arr[i][j]);
        }
        printf("\n");  
    }
    double sum_main_diag = 0;
    for (int i = 0; i < N; i++) {
        sum_main_diag += arr[i][i];
    }
    printf("Сумма главной: %lf\n", sum_main_diag);
    double sum_secondary_diag = 0;
    for (int i = 0; i < N; i++) {
        sum_secondary_diag += arr[i][N - 1 - i];
    }
    printf("Сумма побочной: %lf\n", sum_secondary_diag);
    return 0;
}
