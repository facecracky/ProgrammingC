#include <stdio.h>

int main() {
    int N = 3;
    double arr[N][N];
    for(int i = 0; i < N; i++){
        for(int j = 0; j < N; j++){
            //scanf("%lf", &(arr[i][j]));
            arr[i][j] = 1.0*(i*3 + j);
        }
    }
    double sum = 0;
    for (int i = 0; i < N; i++){
        sum += arr[i][i];
    }
    for(int i = 0; i < N; i++){
        for(int j = 0; j < N; j++){
            printf("%lf", arr[i][j]);
        }
    }
    double sum2 = 0;
    for (int i = 0; i < N; i++){
        sum2 += arr[1][3] + arr[2][2] + arr[3][1];
    }
    printf("%lf", sum);
    printf("%lf", sum2);
    return 0;
}
