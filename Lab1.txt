#include <stdio.h>
#include <math.h>

int main() {
    int choice;

    do {
        double a, b, c;
        double disk, x1, x2;

        printf("Введите a: ");
        scanf("%lf", &a);
        printf("Введите b: ");
        scanf("%lf", &b);
        printf("Введите c: ");
        scanf("%lf", &c);

        disk = b * b - 4 * a * c;

        if (disk >= 0) {
            x1 = (-b + sqrt(disk)) / (2 * a);
            x2 = (-b - sqrt(disk)) / (2 * a);

            printf("Корни:\n");
            printf("x1 = %lf \n", x1);
            printf("x2 = %lf \n", x2);
        } else {
            printf("Нет действительных корней.\n");
        }

        printf("Введите 1, чтобы вычислить корни заново, или 0 для выхода: ");
        scanf("%d", &choice);

    } while (choice == 1);

    printf("Программа завершена.\n");
    return 0;
}
