#include <stdio.h>
#include <math.h>

int main() {

    double a, b, c;
    double disk, x1, x2;

    printf("введи a: ");
    scanf("%lf", &a);
    printf(" b: ");
    scanf("%lf", &b);
    printf(" c: ");
    scanf("%lf", &c);

    disk = b * b - 4 * a * c;

    if (disk >= 0) {
        x1 = (-b + sqrt(disk)) / (2 * a);
        x2 = (-b - sqrt(disk)) / (2 * a);

        printf("корни:\n");
        printf("x1 = %lf \n", x1);
        printf("x2 = %lf \n", x2);
    } else {

        printf("нет решения \n");
    }

    return 0;
}
