#include <stdio.h>
#include <math.h>

int kort(double a, double b, double c);
double p(double a, double b, double c);
double s(double a, double b, double c);

int main() {
    double a, b, c;

    printf("Введите стороны треугольника (a b c): ");
    scanf("%lf %lf %lf", &a, &b, &c);

    if (!kort(a, b, c)) {
        printf("Ошибка: это не треугольник!\n");
        return 1;
    }

    printf("Периметр: %.2f\n", p(a, b, c));
    printf("Площадь: %.2f\n", s(a, b, c));

    return 0;
}

int kort(double a, double b, double c) {
    return (a + b > c) && (a + c > b) && (b + c > a) && (a > 0) && (b > 0) && (c > 0);
}

double p(double a, double b, double c) {
    return a + b + c;
}

double s(double a, double b, double c) {
    double py = p(a, b, c) / 2;
    return sqrt(py * (py - a) * (py - b) * (py - c));
}
